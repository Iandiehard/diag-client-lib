39 uds_transport::UdsTransportProtocolMgr::ConnectionResult DoipTcpChannel::ConnectToHost(
40  uds_transport::UdsMessageConstPtr message) {
41  uds_transport::UdsTransportProtocolMgr::ConnectionResult ret_val{
42  uds_transport::UdsTransportProtocolMgr::ConnectionResult::kConnectionFailed};
43 
44  uds_transport::UdsMessage::IpAddress const kHostIpAddress{message->GetHostIpAddress()};
45  uds_transport::UdsMessage::PortNumber const kHostPortNumber{message->GetHostPortNumber()};
46  // Initiate connecting to server
47  if (tcp_socket_handler_.ConnectToHost(kHostIpAddress, kHostPortNumber)) {
48  // Once connected, Send routing activation req and get response
49  ret_val = tcp_channel_handler_.SendRoutingActivationRequest(std::move(message));
50  } else { // failure
51  logger::DoipClientLogger::GetDiagClientLogger().GetLogger().LogError(
52  __FILE__, __LINE__, __func__, [&kHostIpAddress, &kHostPortNumber](std::stringstream &msg) {
53  msg << "Doip Tcp socket connect failed for remote endpoints : "
54  << "<Ip: " << kHostIpAddress << ", Port: " << kHostPortNumber << ">";
55  });
56  }
57  return ret_val;
58 }

60 uds_transport::UdsTransportProtocolMgr::DisconnectionResult DoipTcpChannel::DisconnectFromHost() {
61  uds_transport::UdsTransportProtocolMgr::DisconnectionResult ret_val{
62  uds_transport::UdsTransportProtocolMgr::DisconnectionResult::kDisconnectionFailed};
63  if (tcp_socket_handler_.DisconnectFromHost()) {
64  if (tcp_channel_handler_.IsRoutingActivated()) {
65  // Reset the handler
66  tcp_channel_handler_.Reset();
67  }
68  ret_val = uds_transport::UdsTransportProtocolMgr::DisconnectionResult::kDisconnectionOk;
69  }
70  return ret_val;
71 }
