41 uds_transport::UdsTransportProtocolMgr::ConnectionResult DoipTcpChannel::ConnectToHost(
42  uds_transport::UdsMessageConstPtr message) {
43  uds_transport::UdsTransportProtocolMgr::ConnectionResult ret_val{
44  uds_transport::UdsTransportProtocolMgr::ConnectionResult::kConnectionFailed};
45 
46  uds_transport::UdsMessage::IpAddress const kHostIpAddress{message->GetHostIpAddress()};
47  uds_transport::UdsMessage::PortNumber const kHostPortNumber{message->GetHostPortNumber()};
48  // Initiate connecting to server
49  if (tcp_socket_handler_.ConnectToHost(kHostIpAddress, kHostPortNumber)) {
50  // Once connected, Send routing activation req and get response
51  ret_val = tcp_channel_handler_.SendRoutingActivationRequest(std::move(message));
52  } else { // failure
53  logger::DoipClientLogger::GetDiagClientLogger().GetLogger().LogError(
54  FILE_NAME, __LINE__, __func__, [&kHostIpAddress, &kHostPortNumber](std::stringstream &msg) {
55  msg << "Doip Tcp socket connect failed for remote endpoints : "
56  << "<Ip: " << kHostIpAddress << ", Port: " << kHostPortNumber << ">";
57  });
58  }
59  return ret_val;
60 }

62 uds_transport::UdsTransportProtocolMgr::DisconnectionResult DoipTcpChannel::DisconnectFromHost() {
63  uds_transport::UdsTransportProtocolMgr::DisconnectionResult ret_val{
64  uds_transport::UdsTransportProtocolMgr::DisconnectionResult::kDisconnectionFailed};
65  if (tcp_socket_handler_.DisconnectFromHost()) {
66  if (tcp_channel_handler_.IsRoutingActivated()) {
67  // Reset the handler
68  tcp_channel_handler_.Reset();
69  }
70  ret_val = uds_transport::UdsTransportProtocolMgr::DisconnectionResult::kDisconnectionOk;
71  }
72  return ret_val;
