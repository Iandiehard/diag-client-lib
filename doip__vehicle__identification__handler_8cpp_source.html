133 void CreateDoipGenericHeader(std::vector<std::uint8_t> &doip_header_buffer, std::uint16_t payload_type,
134  std::uint32_t payload_len) {
135  doip_header_buffer.emplace_back(kDoip_ProtocolVersion);
136  doip_header_buffer.emplace_back(~(static_cast<std::uint8_t>(kDoip_ProtocolVersion)));
137  doip_header_buffer.emplace_back(static_cast<std::uint8_t>((payload_type & 0xFF00) >> 8));
138  doip_header_buffer.emplace_back(static_cast<std::uint8_t>(payload_type & 0x00FF));
139  doip_header_buffer.emplace_back(static_cast<std::uint8_t>((payload_len & 0xFF000000) >> 24));
140  doip_header_buffer.emplace_back(static_cast<std::uint8_t>((payload_len & 0x00FF0000) >> 16));
141  doip_header_buffer.emplace_back(static_cast<std::uint8_t>((payload_len & 0x0000FF00) >> 8));
142  doip_header_buffer.emplace_back(static_cast<std::uint8_t>(payload_len & 0x000000FF));
143 }

151 auto GetVehicleIdentificationPayloadType(std::uint8_t preselection_mode) noexcept -> VehiclePayloadType {
152  std::uint16_t vehicle_identification_type{0u};
153  std::uint8_t vehicle_identification_length{0u};
154  switch (preselection_mode) {
155  case 0U:
156  vehicle_identification_type = kDoip_VehicleIdentification_ReqType;
157  vehicle_identification_length = kDoip_VehicleIdentification_ReqLen;
158  break;
159  case 1U:
160  vehicle_identification_type = kDoip_VehicleIdentificationVIN_ReqType;
161  vehicle_identification_length = kDoip_VehicleIdentificationVIN_ReqLen;
162  break;
163  case 2U:
164  vehicle_identification_type = kDoip_VehicleIdentificationEID_ReqType;
165  vehicle_identification_length = kDoip_VehicleIdentificationEID_ReqLen;
166  break;
167  default:
168  break;
169  }
170  return VehiclePayloadType{vehicle_identification_type, vehicle_identification_length};
171 }

270 auto VehicleIdentificationHandler::HandleVehicleIdentificationRequest(
271  uds_transport::UdsMessageConstPtr vehicle_identification_request) noexcept
272  -> uds_transport::UdsTransportProtocolMgr::TransmissionResult {
273  uds_transport::UdsTransportProtocolMgr::TransmissionResult ret_val{
274  uds_transport::UdsTransportProtocolMgr::TransmissionResult::kTransmitFailed};
275  if (handler_impl_->GetStateContext().GetActiveState().GetState() == VehicleIdentificationState::kIdle) {
276  if (SendVehicleIdentificationRequest(std::move(vehicle_identification_request)) ==
277  uds_transport::UdsTransportProtocolMgr::TransmissionResult::kTransmitOk) {
278  ret_val = uds_transport::UdsTransportProtocolMgr::TransmissionResult::kTransmitOk;
279 
280  handler_impl_->GetStateContext().TransitionTo(VehicleIdentificationState::kWaitForVehicleIdentificationRes);
281  // Wait for 2 sec to collect all the vehicle identification response
282  handler_impl_->GetSyncTimer().WaitForTimeout(
283  [&]() { handler_impl_->GetStateContext().TransitionTo(VehicleIdentificationState::kDoIPCtrlTimeout); },
284  [&]() {
285  // do nothing
286  },
287  std::chrono::milliseconds{kDoIPCtrl});
288  handler_impl_->GetStateContext().TransitionTo(VehicleIdentificationState::kIdle);
289  } else {
290  // failed, do nothing
291  }
292  } else {
293  // not free
294  }
295  return ret_val;
296 }

298 void VehicleIdentificationHandler::ProcessVehicleIdentificationResponse(DoipMessage &doip_payload) noexcept {
299  if (handler_impl_->GetStateContext().GetActiveState().GetState() ==
300  VehicleIdentificationState::kWaitForVehicleIdentificationRes) {
301  // Deserialize data to indicate to upper layer
302  std::pair<uds_transport::UdsTransportProtocolMgr::IndicationResult, uds_transport::UdsMessagePtr> ret_val{
303  handler_impl_->GetDoipChannel().IndicateMessage(
304  static_cast<uds_transport::UdsMessage::Address>(0U), static_cast<uds_transport::UdsMessage::Address>(0U),
305  uds_transport::UdsMessage::TargetAddressType::kPhysical, 0U, doip_payload.GetPayload().size(), 0U,
306  "DoIPUdp", doip_payload.GetPayload())};
307  if ((ret_val.first == uds_transport::UdsTransportProtocolMgr::IndicationResult::kIndicationOk) &&
308  (ret_val.second != nullptr)) {
309  // Add meta info about ip address
310  uds_transport::UdsMessage::MetaInfoMap meta_info_map{
311  {"kRemoteIpAddress", std::string{doip_payload.GetHostIpAddress()}}};
312  ret_val.second->AddMetaInfo(std::make_shared<uds_transport::UdsMessage::MetaInfoMap>(meta_info_map));
313  // copy to application buffer
314  (void) std::copy(doip_payload.GetPayload().begin(), doip_payload.GetPayload().end(),
315  ret_val.second->GetPayload().begin());
316  handler_impl_->GetDoipChannel().HandleMessage(std::move(ret_val.second));
317  }
318  } else {
319  // ignore
320  }
321 }

323 auto VehicleIdentificationHandler::SendVehicleIdentificationRequest(
324  uds_transport::UdsMessageConstPtr vehicle_identification_request) noexcept
325  -> uds_transport::UdsTransportProtocolMgr::TransmissionResult {
326  uds_transport::UdsTransportProtocolMgr::TransmissionResult ret_val{
327  uds_transport::UdsTransportProtocolMgr::TransmissionResult::kTransmitFailed};
328  UdpMessagePtr doip_vehicle_identification_req{std::make_unique<UdpMessage>(
329  vehicle_identification_request->GetHostIpAddress(), vehicle_identification_request->GetHostPortNumber())};
330 
331  // Get preselection mode
332  std::uint8_t preselection_mode{vehicle_identification_request->GetPayload()[BYTE_POS_ONE]};
333  // Get the payload type & length from preselection mode
334  VehiclePayloadType const doip_vehicle_payload_type{GetVehicleIdentificationPayloadType(preselection_mode)};
335 
336  // create header
337  CreateDoipGenericHeader(doip_vehicle_identification_req->GetTxBuffer(), doip_vehicle_payload_type.first,
338  doip_vehicle_payload_type.second);
339  // Copy only if containing VIN / EID
340  if (doip_vehicle_payload_type.first != kDoip_VehicleIdentification_ReqType) {
341  doip_vehicle_identification_req->GetTxBuffer().insert(
342  doip_vehicle_identification_req->GetTxBuffer().begin() + kDoipheadrSize,
343  vehicle_identification_request->GetPayload().begin() + 2U, vehicle_identification_request->GetPayload().end());
344  }
345  if (handler_impl_->GetSocketHandler().Transmit(std::move(doip_vehicle_identification_req))) {
346  ret_val = uds_transport::UdsTransportProtocolMgr::TransmissionResult::kTransmitOk;
347  }
348  return ret_val;
349 }
